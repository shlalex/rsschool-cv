const arr = [6,3,7,9,2,1,4,5,8];

function bubbleSort (arr) {
     for (let i = 0, endI = arr.length - 1; i < endI; i++) {
          let flag = false;
         for (let j = 0, endJ = endI - i; j < endJ; j++) {
             if (arr[j] > arr[j + 1]) {
                 [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
                 flag = true;
             }
         }
         if (!flag) break;
     }
     return arr;
 }
